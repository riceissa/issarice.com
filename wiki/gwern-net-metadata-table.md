# Source

To reproduce the table, first clone the gwern.net repository:

    git clone https://github.com/gwern/gwern.net.git

Then run:

    find . -iname '*.page' | while read filename
    do
        cat "$filename" | ./metadata.py "$filename" >> temp
    done

Where `metadata.py` is:

    #!/usr/bin/env python3

    import sys
    import urllib.parse


    md = {}
    md_keys = [
        "title",
        "created",
        "importance",
        "status",
        "confidence",
    ]
    for line in sys.stdin:
        for k in md_keys:
            if k not in md and line.startswith(k + ':'):
                md[k] = line[len(k + ": "):].strip()
    if "title" not in md:
        md["title"] = sys.argv[1]
    md["linked_title"] = "[{}]({})".format(md["title"], "https://gwern.net/" + urllib.parse.quote(sys.argv[1][2:-5]))

    print("|" + "|".join([
          md.get(k, "")
          for k in ["linked_title", "created", "importance", "status",
                    "confidence"]
          ]) + "|")
