    find . -iname '*.page' | while read filename; do cat "$filename" | ./metadata.py "$filename" >> gwern_metadata.md; done

Where `metadata.py` is:

    #!/usr/bin/env python3

    import sys


    md = {}
    md_keys = [
        "title",
        "created",
        "importance",
        "status",
        "confidence",
    ]
    for line in sys.stdin:
        for k in md_keys:
            if k not in md and line.startswith(k + ':'):
                md[k] = line[len(k + ": "):].strip()
    if "title" not in md:
        md["title"] = sys.argv[1]

    print("|" + "|".join([md.get(k, "") for k in md_keys]) + "|")
